3   import eu.fbk.utils.gson.JSONLabel;
4   
5   import java.io.IOException;

13  public class ClassesTest {
14  
15      public static void main(String[] args) {
16          final ClassLoader loader = Thread.currentThread()
17                  .getContextClassLoader();
18          try {
19  
20              ClassPath classpath = ClassPath.from(loader); // scans the class path used by classloader
21              for (ClassPath.ClassInfo classInfo : classpath.getAllClasses()) {
22                  if (!classInfo.getSimpleName().endsWith("_")) {
23                      try {
24                          Class<?> myClass = classInfo.load();
25                          if (myClass.isAnnotationPresent(JSONLabel.class)) {
26                              JSONLabel JsonAnnotation = myClass.getAnnotation(JSONLabel.class);
27                              Class<?>[] serializerClasses = JsonAnnotation.serializer();
28                              for (Class<?> serializerClass : serializerClasses) {
29                                  if (JsonSerializer.class.isAssignableFrom(serializerClass)) {
30                                      for (Type type : serializerClass.getGenericInterfaces()) {
31                                          if (type instanceof ParameterizedType && ((ParameterizedType) type).getRawType()
32                                                  .equals(JsonSerializer.class)) {
33                                              System.out.println(((ParameterizedType) type).getActualTypeArguments()[0]);
34                                          }
35  
36                                      }
37                                  }
38                              }
39  
40                          }
41                      } catch (Throwable e) {
42  //                        e.printStackTrace();
43                      }
44                  }
45              }
46          } catch (IOException e) {
47              e.printStackTrace();
48          }
49  
50      }
51  }
