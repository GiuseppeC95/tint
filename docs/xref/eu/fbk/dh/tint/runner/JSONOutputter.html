37   * Output an Annotation to human readable JSON.
38   * This is not a lossless operation; for more strict serialization,

49      public static String cleanJSON(String s) {
50          return s
51                  .replace("\\", "\\\\")
52                  .replace("\b", "\\b")
53                  .replace("\f", "\\f")
54                  .replace("\n", "\\n")
55                  .replace("\r", "\\r")
56                  .replace("\t", "\\t")
57                  .replace("\"", "\\\"");
58      }

60      /** {@inheritDoc} */
61  //    @SuppressWarnings("RedundantCast")  // It's lying; we need the "redundant" casts (as of 2014-09-08)
62      @Override
63      public void print(Annotation doc, OutputStream target, Options options) throws IOException {
64          JSONWriter l0 = new JSONWriter(new PrintWriter(target), options);
65  
66          l0.object(l1 -> {
67  
68              // Add annotations attached to a Document
69              l1.set("docId", doc.get(CoreAnnotations.DocIDAnnotation.class));
70              l1.set("docDate", doc.get(CoreAnnotations.DocDateAnnotation.class));
71              l1.set("docSourceType", doc.get(CoreAnnotations.DocSourceTypeAnnotation.class));
72              l1.set("docType", doc.get(CoreAnnotations.DocTypeAnnotation.class));
73              l1.set("author", doc.get(CoreAnnotations.AuthorAnnotation.class));
74              l1.set("location", doc.get(CoreAnnotations.LocationAnnotation.class));
75              if (options.includeText) {
76                  l1.set("text", doc.get(CoreAnnotations.TextAnnotation.class));
77              }
78  
79              // Add sentences
80              if (doc.get(CoreAnnotations.SentencesAnnotation.class) != null) {
81                  l1.set("sentences", doc.get(CoreAnnotations.SentencesAnnotation.class).stream()
82                          .map(sentence -> (Consumer<Writer>) (Writer l2) -> {
83                              // Add a single sentence
84                              // (metadata)
85                              l2.set("id", sentence.get(CoreAnnotations.SentenceIDAnnotation.class));
86                              l2.set("index", sentence.get(CoreAnnotations.SentenceIndexAnnotation.class));
87                              l2.set("line", sentence.get(CoreAnnotations.LineNumberAnnotation.class));
88                              // (constituency tree)
89                              StringWriter treeStrWriter = new StringWriter();
90                              TreePrint treePrinter = options.constituentTreePrinter;
91                              if (treePrinter == AnnotationOutputter.DEFAULT_CONSTITUENT_TREE_PRINTER) {
92                                  // note the '==' -- we're overwriting the default, but only if it was not explicitly set otherwise
93                                  treePrinter = new TreePrint("oneline");
94                              }
95                              treePrinter.printTree(sentence.get(TreeCoreAnnotations.TreeAnnotation.class),
96                                      new PrintWriter(treeStrWriter, true));
97                              l2.set("parse", treeStrWriter.toString().trim());  // strip the trailing newline
98                              // (dependency trees)
99                              l2.set("basic-dependencies", buildDependencyTree(
100                                     sentence.get(SemanticGraphCoreAnnotations.BasicDependenciesAnnotation.class)));
101                             l2.set("collapsed-dependencies", buildDependencyTree(
102                                     sentence.get(SemanticGraphCoreAnnotations.CollapsedDependenciesAnnotation.class)));
103                             l2.set("collapsed-ccprocessed-dependencies", buildDependencyTree(sentence.get(
104                                     SemanticGraphCoreAnnotations.CollapsedCCProcessedDependenciesAnnotation.class)));
105                             // (sentiment)
106                             Tree sentimentTree = sentence.get(SentimentCoreAnnotations.SentimentAnnotatedTree.class);
107                             if (sentimentTree != null) {
108                                 int sentiment = RNNCoreAnnotations.getPredictedClass(sentimentTree);
109                                 String sentimentClass = sentence.get(SentimentCoreAnnotations.SentimentClass.class);
110                                 l2.set("sentimentValue", Integer.toString(sentiment));
111                                 l2.set("sentiment", sentimentClass.replaceAll(" ", ""));
112                             }
113                             // (openie)
114                             Collection<RelationTriple> openIETriples = sentence
115                                     .get(NaturalLogicAnnotations.RelationTriplesAnnotation.class);
116                             if (openIETriples != null) {
117                                 l2.set("openie", openIETriples.stream()
118                                         .map(triple -> (Consumer<Writer>) (Writer tripleWriter) -> {
119                                             tripleWriter.set("subject", triple.subjectGloss());
120                                             tripleWriter.set("subjectSpan", Span.fromPair(triple.subjectTokenSpan()));
121                                             tripleWriter.set("relation", triple.relationGloss());
122                                             tripleWriter.set("relationSpan", Span.fromPair(triple.relationTokenSpan()));
123                                             tripleWriter.set("object", triple.objectGloss());
124                                             tripleWriter.set("objectSpan", Span.fromPair(triple.objectTokenSpan()));
125                                         }));
126                             }
127 
128                             // (add tokens)
129                             if (sentence.get(CoreAnnotations.TokensAnnotation.class) != null) {
130                                 l2.set("tokens", sentence.get(CoreAnnotations.TokensAnnotation.class).stream()
131                                         .map(token -> (Consumer<Writer>) (Writer l3) -> {
132                                             // Add a single token
133                                             l3.set("index", token.index());
134                                             l3.set("word", token.word());
135                                             l3.set("originalText", token.originalText());
136                                             l3.set("lemma", token.lemma());
137                                             l3.set("full_morpho", token.get(DigiMorphAnnotations.MorphoAnnotation.class));
138                                             l3.set("characterOffsetBegin", token.beginPosition());
139                                             l3.set("characterOffsetEnd", token.endPosition());
140                                             l3.set("pos", token.tag());
141                                             l3.set("ner", token.ner());
142 
143                                             // todo: quickly added by Alessio
144                                             l3.set("tense", token.get(TenseAnnotations.TenseAnnotation.class));
145 
146 //                                            String linkedPage = null;
147 //                                            LinkingTag linkingTag = token
148 //                                                    .get(PikesAnnotations.DBpediaSpotlightAnnotation.class);
149 //                                            if (linkingTag != null) {
150 //                                                linkedPage = linkingTag.getPage();
151 //                                            }
152 //                                            l3.set("link", linkedPage);
153 
154                                             l3.set("normalizedNER", token.get(
155                                                     CoreAnnotations.NormalizedNamedEntityTagAnnotation.class));
156                                             l3.set("speaker", token.get(CoreAnnotations.SpeakerAnnotation.class));
157                                             l3.set("truecase", token.get(CoreAnnotations.TrueCaseAnnotation.class));
158                                             l3.set("truecaseText",
159                                                     token.get(CoreAnnotations.TrueCaseTextAnnotation.class));
160                                             l3.set("before", token.get(CoreAnnotations.BeforeAnnotation.class));
161                                             l3.set("after", token.get(CoreAnnotations.AfterAnnotation.class));
162                                             // Timex
163                                             Timex time = token.get(TimeAnnotations.TimexAnnotation.class);
164                                             if (time != null) {
165                                                 l3.set("timex", (Consumer<Writer>) l4 -> {
166                                                     l4.set("tid", time.tid());
167                                                     l4.set("type", time.timexType());
168                                                     l4.set("value", time.value());
169                                                     l4.set("altValue", time.altVal());
170                                                 });
171                                             }
172                                         }));
173                             }
174                         }));
175             }
176 
177             // Add coref values
178             if (doc.get(CorefCoreAnnotations.CorefChainAnnotation.class) != null) {
179                 Map<Integer, CorefChain> corefChains =
180                         doc.get(CorefCoreAnnotations.CorefChainAnnotation.class);
181                 if (corefChains != null) {
182                     l1.set("corefs", (Consumer<Writer>) chainWriter -> {
183                         for (CorefChain chain : corefChains.values()) {
184                             CorefChain.CorefMention representative = chain.getRepresentativeMention();
185                             chainWriter.set(Integer.toString(chain.getChainID()),
186                                     chain.getMentionsInTextualOrder().stream()
187                                             .map(mention -> (Consumer<Writer>) (Writer mentionWriter) -> {
188                                                 mentionWriter.set("id", mention.mentionID);
189                                                 mentionWriter.set("text", Sentence
190                                                         .listToOriginalTextString(
191                                                                 doc.get(CoreAnnotations.SentencesAnnotation.class)
192                                                                         .get(mention.sentNum - 1)
193                                                                         .get(CoreAnnotations.TokensAnnotation.class)
194                                                                         .subList(mention.startIndex - 1,
195                                                                                 mention.endIndex - 1)).trim());
196                                                 mentionWriter.set("type", mention.mentionType);
197                                                 mentionWriter.set("number", mention.number);
198                                                 mentionWriter.set("gender", mention.gender);
199                                                 mentionWriter.set("animacy", mention.animacy);
200                                                 mentionWriter.set("startIndex", mention.startIndex);
201                                                 mentionWriter.set("endIndex", mention.endIndex);
202                                                 mentionWriter.set("sentNum", mention.sentNum);
203                                                 mentionWriter.set("position",
204                                                         Arrays.stream(mention.position.elems()).boxed()
205                                                                 .collect(Collectors.toList()));
206                                                 mentionWriter.set("isRepresentativeMention", mention == representative);
207                                             }));
208                         }
209                     });
210                 }
211             }
212         });
213 
214         l0.writer.flush();  // flush
215     }

217     /**
218      * Convert a dependency graph to a format expected as input to {@link Writer#set(String, Object)}.
219      */
220     @SuppressWarnings("RedundantCast")  // It's lying; we need the "redundant" casts (as of 2014-09-08)
221     private static Object buildDependencyTree(SemanticGraph graph) {
222         if (graph != null) {
223             return Stream.concat(
224                     // Roots
225                     graph.getRoots().stream().map((IndexedWord root) -> (Consumer<Writer>) dep -> {
226                         dep.set("dep", "ROOT");
227                         dep.set("governor", 0);
228                         dep.set("governorGloss", "ROOT");
229                         dep.set("dependent", root.index());
230                         dep.set("dependentGloss", root.word());
231                     }),
232                     // Regular edges
233                     graph.edgeListSorted().stream().map((SemanticGraphEdge edge) -> (Consumer<Writer>) (Writer dep) -> {
234                         dep.set("dep", edge.getRelation().toString());
235                         dep.set("governor", edge.getGovernor().index());
236                         dep.set("governorGloss", edge.getGovernor().word());
237                         dep.set("dependent", edge.getDependent().index());
238                         dep.set("dependentGloss", edge.getDependent().word());
239                     })
240             );
241         } else {
242             return null;
243         }
244     }

264     /**
265      * <p>Our very own little JSON writing class.
266      * For usage, see the test cases in JSONOutputterTest.</p>
267      * <p>
268      * <p>For the love of all that is holy, don't try to write JSON multithreaded.
269      * It should go without saying that this is not threadsafe.</p>
270      */
271     protected static class JSONWriter {
272 
273         private final PrintWriter writer;
274         private final Options options;
275 
276         private JSONWriter(PrintWriter writer, Options options) {
277             this.writer = writer;
278             this.options = options;
279         }
280 
281         @SuppressWarnings({ "unchecked", "UnnecessaryBoxing" })
282         private void routeObject(int indent, Object value) {
283             if (value instanceof String) {
284                 // Case: simple string (this is easy!)
285                 writer.write("\"");
286                 writer.write(cleanJSON(value.toString()));
287                 writer.write("\"");
288             } else if (value instanceof Collection) {
289                 // Case: collection
290                 writer.write("[");
291                 newline();
292                 Iterator<Object> elems = ((Collection<Object>) value).iterator();
293                 while (elems.hasNext()) {
294                     indent(indent + 1);
295                     routeObject(indent + 1, elems.next());
296                     if (elems.hasNext()) {
297                         writer.write(",");
298                     }
299                     newline();
300                 }
301                 indent(indent);
302                 writer.write("]");
303             } else if (value instanceof Enum) {
304                 // Case: enumeration constant
305                 writer.write("\"");
306                 writer.write(cleanJSON(((Enum) value).name()));
307                 writer.write("\"");
308             } else if (value instanceof Pair) {
309                 routeObject(indent, Arrays.asList(((Pair) value).first, ((Pair) value).second));
310             } else if (value instanceof Span) {
311                 writer.write("[");
312                 writer.write(Integer.toString(((Span) value).start()));
313                 writer.write(",");
314                 space();
315                 writer.write(Integer.toString(((Span) value).end()));
316                 writer.write("]");
317             } else if (value instanceof Consumer) {
318                 object(indent, (Consumer<Writer>) value);
319             } else if (value instanceof Stream) {
320                 routeObject(indent, ((Stream) value).collect(Collectors.toList()));
321             } else if (value.getClass().isArray()) {
322                 // Arrays make life miserable in Java
323                 Class<?> componentType = value.getClass().getComponentType();
324                 if (componentType.isPrimitive()) {
325                     if (int.class.isAssignableFrom(componentType)) {
326                         ArrayList<Integer> lst = new ArrayList<>();
327                         for (int elem : ((int[]) value)) {
328                             lst.add(elem);
329                         }
330                         routeObject(indent, lst);
331                     } else if (short.class.isAssignableFrom(componentType)) {
332                         ArrayList<Short> lst = new ArrayList<>();
333                         for (short elem : ((short[]) value)) {
334                             lst.add(elem);
335                         }
336                         routeObject(indent, lst);
337                     } else if (byte.class.isAssignableFrom(componentType)) {
338                         ArrayList<Byte> lst = new ArrayList<>();
339                         for (byte elem : ((byte[]) value)) {
340                             lst.add(elem);
341                         }
342                         routeObject(indent, lst);
343                     } else if (long.class.isAssignableFrom(componentType)) {
344                         ArrayList<Long> lst = new ArrayList<>();
345                         for (long elem : ((long[]) value)) {
346                             lst.add(elem);
347                         }
348                         routeObject(indent, lst);
349                     } else if (char.class.isAssignableFrom(componentType)) {
350                         ArrayList<Character> lst = new ArrayList<>();
351                         for (char elem : ((char[]) value)) {
352                             lst.add(elem);
353                         }
354                         routeObject(indent, lst);
355                     } else if (float.class.isAssignableFrom(componentType)) {
356                         ArrayList<Float> lst = new ArrayList<>();
357                         for (float elem : ((float[]) value)) {
358                             lst.add(elem);
359                         }
360                         routeObject(indent, lst);
361                     } else if (double.class.isAssignableFrom(componentType)) {
362                         ArrayList<Double> lst = new ArrayList<>();
363                         for (double elem : ((double[]) value)) {
364                             lst.add(elem);
365                         }
366                         routeObject(indent, lst);
367                     } else if (boolean.class.isAssignableFrom(componentType)) {
368                         ArrayList<Boolean> lst = new ArrayList<>();
369                         for (boolean elem : ((boolean[]) value)) {
370                             lst.add(elem);
371                         }
372                         routeObject(indent, lst);
373                     } else {
374                         throw new IllegalStateException("Unhandled primitive type in array: " + componentType);
375                     }
376                 } else {
377                     routeObject(indent, Arrays.asList((Object[]) value));
378                 }
379             } else if (value instanceof Integer) {
380                 writer.write(Integer.toString((Integer) value));
381             } else if (value instanceof Short) {
382                 writer.write(Short.toString((Short) value));
383             } else if (value instanceof Byte) {
384                 writer.write(Byte.toString((Byte) value));
385             } else if (value instanceof Long) {
386                 writer.write(Long.toString((Long) value));
387             } else if (value instanceof Character) {
388                 writer.write(Character.toString((Character) value));
389             } else if (value instanceof Float) {
390                 writer.write(new DecimalFormat("0.#######").format(value));
391             } else if (value instanceof Double) {
392                 writer.write(new DecimalFormat("0.##############").format(value));
393             } else if (value instanceof Boolean) {
394                 writer.write(Boolean.toString((Boolean) value));
395             } else if (int.class.isAssignableFrom(value.getClass())) {
396                 routeObject(indent, Integer.valueOf((int) value));
397             } else if (short.class.isAssignableFrom(value.getClass())) {
398                 routeObject(indent, Short.valueOf((short) value));
399             } else if (byte.class.isAssignableFrom(value.getClass())) {
400                 routeObject(indent, Byte.valueOf((byte) value));
401             } else if (long.class.isAssignableFrom(value.getClass())) {
402                 routeObject(indent, Long.valueOf((long) value));
403             } else if (char.class.isAssignableFrom(value.getClass())) {
404                 routeObject(indent, Character.valueOf((char) value));
405             } else if (float.class.isAssignableFrom(value.getClass())) {
406                 routeObject(indent, Float.valueOf((float) value));
407             } else if (double.class.isAssignableFrom(value.getClass())) {
408                 routeObject(indent, Double.valueOf((double) value));
409             } else if (boolean.class.isAssignableFrom(value.getClass())) {
410                 routeObject(indent, Boolean.valueOf((boolean) value));
411             } else {
412                 throw new RuntimeException("Unknown object to serialize: " + value);
413             }
414         }
415 
416         public void object(int indent, Consumer<Writer> callback) {
417             writer.write("{");
418             final Pointer<Boolean> firstCall = new Pointer<>(true);
419             callback.accept((key, value) -> {
420                 if (key != null && value != null) {
421                     // First call overhead
422                     if (!firstCall.dereference().get()) {
423                         writer.write(",");
424                     }
425                     firstCall.set(false);
426                     // Write the key
427                     newline();
428                     indent(indent + 1);
429                     writer.write("\"");
430                     writer.write(cleanJSON(key));
431                     writer.write("\":");
432                     space();
433                     // Write the value
434                     routeObject(indent + 1, value);
435                 }
436             });
437             newline();
438             indent(indent);
439             writer.write("}");
440         }
441 
442         public void object(Consumer<Writer> callback) {
443             object(0, callback);
444         }
445 
446         private void indent(int num) {
447             if (options.pretty) {
448                 for (int i = 0; i < num; ++i) {
449                     writer.write(INDENT_CHAR);
450                 }
451             }
452         }
453 
454         private void space() {
455             if (options.pretty) {
456                 writer.write(" ");
457             }
458         }
459 
460         private void newline() {
461             if (options.pretty) {
462                 writer.write("\n");
463             }
464         }
465 
466         public static String objectToJSON(Consumer<Writer> callback) {
467             OutputStream os = new ByteArrayOutputStream();
468             PrintWriter out = new PrintWriter(os);
469             new JSONOutputter.JSONWriter(out, new Options()).object(callback);
470             out.close();
471             return os.toString();
472         }

475     /**
476      * A tiny little functional interface for writing a (key, value) pair.
477      * The key should always be a String, the value can be either a String,
478      * a Collection of valid values, or a Callback taking a Writer (this is how
479      * we represent objects while creating JSON).
480      */
481     @FunctionalInterface
482     protected interface Writer {
483 
484         /**
485          * Set a (key, value) pair in a JSON object.
486          * Note that if either the key or the value is null, nothing will be set.
487          *
488          * @param key   The key of the object.
489          * @param value The value of the object.
490          */
491         void set(String key, Object value);
492     }
