15  public class TintPipeline {
16  
17      private static final Logger LOGGER = LoggerFactory.getLogger(TintPipeline.class);
18      StanfordCoreNLP pipeline = null;
19      String documentDate = null;
20      Properties props = new Properties();
21  
22      public void load() {
23          if (pipeline == null) {
24              pipeline = new StanfordCoreNLP(props);
25          }
26      }
27  
28      public void loadDefaultProperties() throws IOException {
29          InputStream configStream = TintRunner.class.getResourceAsStream("/default-config.properties");
30          if (configStream != null) {
31              props.load(configStream);
32          }
33      }
34  
35      public void loadPropertiesFromStream(InputStream stream) throws IOException {
36          props.load(stream);
37      }
38  
39      public void loadPropertiesFromFile(File propsFile) throws IOException {
40          if (propsFile != null) {
41              InputStream configStream = new FileInputStream(propsFile);
42              loadPropertiesFromStream(configStream);
43          }
44      }
45  
46      public void addProperties(Properties properties) {
47          if (properties != null) {
48              props.putAll(properties);
49          }
50      }
51  
52      public String getDocumentDate() {
53          return documentDate;
54      }
55  
56      public void setDocumentDate(String documentDate) {
57          this.documentDate = documentDate;
58      }
59  
60      public Annotation runRaw(String text) {
61          load();
62  
63          Annotation annotation = new Annotation(text);
64          LOGGER.debug("Text: {}", text);
65          if (documentDate != null) {
66              annotation.set(CoreAnnotations.DocDateAnnotation.class, documentDate);
67          }
68          pipeline.annotate(annotation);
69  
70          return annotation;
71      }
72  
73      public Annotation run(InputStream inputStream, OutputStream outputStream, TintRunner.OutputFormat format)
74              throws IOException {
75  
76          Reader reader = new InputStreamReader(inputStream);
77          StringBuilder inputText = new StringBuilder();
78          int i;
79          while ((i = reader.read()) != -1) {
80              inputText.append((char) i);
81          }
82          reader.close();
83          String text = inputText.toString();
84  
85          Annotation annotation = runRaw(text);
86  
87          switch (format) {
88          case CONLL:
89              CoNLLUOutputter.conllUPrint(annotation, outputStream, pipeline);
90              break;
91          case READABLE:
92              TextOutputter.prettyPrint(annotation, outputStream, pipeline);
93              break;
94          case XML:
95              XMLOutputter.xmlPrint(annotation, outputStream, pipeline);
96              break;
97          case JSON:
98              JSONOutputter.jsonPrint(annotation, outputStream, pipeline);
99              break;
100         case TEXTPRO:
101             TextProOutputter.tpPrint(annotation, outputStream, pipeline);
102             break;
103         case NAF:

108 //            pikesPipeline.addToNerMap("LOC", "LOCATION");
109 //            pikesPipeline.annotateStanford(new Properties(), annotation, doc);
