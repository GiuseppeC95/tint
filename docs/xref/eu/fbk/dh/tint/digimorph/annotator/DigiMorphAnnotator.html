1   package eu.fbk.dh.tint.digimorph.annotator;
2   
3   import edu.stanford.nlp.ling.CoreAnnotations;
4   import edu.stanford.nlp.ling.CoreLabel;
5   import edu.stanford.nlp.pipeline.Annotation;
6   import edu.stanford.nlp.pipeline.Annotator;
7   import edu.stanford.nlp.util.CoreMap;
8   import eu.fbk.dh.tint.digimorph.DigiMorph;
9   
10  import java.util.*;
11  
12  /**
13   * Created by giovannimoretti on 18/05/16.
14   */
15  public class DigiMorphAnnotator implements Annotator {
16  
17      DigiMorph dm;
18  
19      public DigiMorphAnnotator(String annotatorName, Properties prop) {
20          String model_path = prop.getProperty(annotatorName + ".model");
21          this.dm = DigiMorphModel.getInstance(model_path);
22      }
23  
24      public void annotate(Annotation annotation) {
25  
26  
27          List<String> token_word = new LinkedList<String>();
28  
29          if (annotation.has(CoreAnnotations.SentencesAnnotation.class)) {
30              for (CoreMap sentence : annotation.get(CoreAnnotations.SentencesAnnotation.class)) {
31                  List<CoreLabel> tokens = sentence.get(CoreAnnotations.TokensAnnotation.class);
32                  for (CoreLabel c : tokens) {
33                      token_word.add(c.word());
34                  }
35              }
36              token_word = dm.getMorphology(token_word);
37              for (CoreMap sentence : annotation.get(CoreAnnotations.SentencesAnnotation.class)) {
38                  List<CoreLabel> tokens = sentence.get(CoreAnnotations.TokensAnnotation.class);
39                  for (CoreLabel c : tokens) {
40                      c.set(CoreAnnotations.MorphoCaseAnnotation.class, token_word.get(0));
41                      token_word.remove(0);
42                  }
43              }
44  
45          }
46      }
47  
48      public Set<Requirement> requirementsSatisfied() {
49          return Collections.singleton(DigiMorphAnnotations.DH_MORPHOLOGY_REQUIREMENT);
50      }
51  
52      public Set<Requirement> requires() {
53          return TOKENIZE_AND_SSPLIT;
54      }
55  
56  }
